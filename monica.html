    /* ===============================

       Highlight numeric metrics in gold

       =============================== */

    function highlightMetrics(){

      document.querySelectorAll('[data-metric]').forEach(el=>{

        const t = el.textContent;

        el.innerHTML = t.replace(/(~?\s?\d+[\d,]*\s?%|\d+[\d,]*\s?\+|\d+k\+?)/gi,

          m=>`<span class="kpi">${m.trim()}</span>`);

      });

    }


    /* ===============================

       setLang(lang):

       - apply EN/DE text

       - swap CV link

       - toggle active state

       - translate badges

       - update Menu / Menü label

       - store preference & reflect ?lang=

       =============================== */

    function setLang(lang){

      const dict = (i18n && i18n[lang]) ? i18n[lang] : i18n.en;


      // normal text nodes

      document.querySelectorAll('[data-i18n]').forEach(el=>{

        const key = el.getAttribute('data-i18n');

        if(dict[key]) el.textContent = dict[key];

      });


      // badges in experience

      document.querySelectorAll('.exp-tags .badge').forEach(b=>{

        const raw = b.getAttribute('data-raw') || b.textContent.trim();

        // store canonical form once so we can re-translate

        if(!b.getAttribute('data-raw')){

          b.setAttribute('data-raw', raw.toLowerCase());

        }

        const canon = b.getAttribute('data-raw'); // e.g. "cross-functional"

        const translated = (badgeI18n[lang] && badgeI18n[lang][canon])

          ? badgeI18n[lang][canon]

          : raw;

        b.textContent = translated;

      });


      // menu label on hamburger

      const menuLabel = document.getElementById('menu-label');

      if (menuLabel && dict.menu) {

        menuLabel.textContent = dict.menu;

      }


      // language pill active state

      const enBtn = document.getElementById('btn-en');

      const deBtn = document.getElementById('btn-de');

      if(enBtn && deBtn){

        enBtn.classList.toggle('active', lang === 'en');

        deBtn.classList.toggle('active', lang === 'de');

      }


      // html lang attr

      document.documentElement.setAttribute('lang', lang);


      // CV links + button text in header/about

      const cvHref = (lang === 'de')

        ? 'assets/CV_MuziChen_DE.pdf'

        : 'assets/CV_MuziChen_EN.pdf';


      const cvTop = document.getElementById('cvTop');

      const cvAbout = document.getElementById('cvAbout');

      if(cvTop){

        cvTop.setAttribute('href', cvHref);

        cvTop.textContent = (lang === 'de') ? '↓ Lebenslauf (PDF)' : '↓ CV (PDF)';

      }

      if(cvAbout){

        cvAbout.setAttribute('href', cvHref);

        cvAbout.textContent = dict.cta_cv || cvAbout.textContent;

      }


      // store preference

      try { localStorage.setItem('lang', lang); } catch(e){}


      // reflect in URL (no reload)

      const url = new URL(window.location);

      url.searchParams.set('lang', lang);

      history.replaceState({}, '', url);


      // refresh metric highlight

      highlightMetrics();

    }


    /* ===============================

       Init after DOM ready:

       - restore lang

       - wire up buttons

       - mobile menu toggle

       - dynamic nav height for dropdown

       =============================== */

    document.addEventListener('DOMContentLoaded', ()=>{

      // choose language

      const qsLang = new URLSearchParams(location.search).get('lang');

      let savedLang = null;

      try { savedLang = localStorage.getItem('lang'); } catch(e){}

      const lang = (qsLang === 'de' || qsLang === 'en') ? qsLang : (savedLang || 'en');

      setLang(lang);


      // attach lang switchers

      const enBtn = document.getElementById('btn-en');

      const deBtn = document.getElementById('btn-de');

      if(enBtn) enBtn.addEventListener('click', ()=>setLang('en'));

      if(deBtn) deBtn.addEventListener('click', ()=>setLang('de'));


      // year in footer

      const y = document.getElementById('year');

      if (y) { y.textContent = new Date().getFullYear(); }


      // mobile menu logic

      const toggleBtn = document.querySelector('.nav-toggle');

      const headerEl = document.querySelector('header.site');

      const topbar = document.querySelector('.topbar');


      // set CSS var --navH for dropdown offset

      const setNavH = () => {

        if(topbar){

          document.documentElement.style.setProperty('--navH', topbar.offsetHeight + 'px');

        }

      };

      setNavH();

      window.addEventListener('resize', ()=>{

        setNavH();

        // close menu when back to desktop

        if(window.innerWidth > 860 && headerEl && toggleBtn){

          headerEl.classList.remove('open');

          toggleBtn.setAttribute('aria-expanded','false');

        }

      });


      if(toggleBtn && headerEl){

        toggleBtn.addEventListener('click', ()=>{

          const isOpen = headerEl.classList.toggle('open');

          toggleBtn.setAttribute('aria-expanded', isOpen ? 'true' : 'false');

        });

      }


      // close dropdown when clicking a nav link (mobile)

      document.querySelectorAll('nav a').forEach(a=>{

        a.addEventListener('click', ()=>{

          if(headerEl && headerEl.classList.contains('open') && toggleBtn){

            headerEl.classList.remove('open');

            toggleBtn.setAttribute('aria-expanded','false');

          }

        });

      });

    });
