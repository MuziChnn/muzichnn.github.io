    // Light/Dark toggle (classless: just swap variables at runtime)

    const btn = document.getElementById('mode');

    const LS_KEY = 'pref-mode';

    const root = document.documentElement;

    const LIGHT = {bg:'#f7f8fb', bgalt:'#ffffff', fg:'#0b1220', muted:'#546178', card:'#ffffff'};

    const DARK  = {bg:'#0b0c10', bgalt:'#111318', fg:'#e8ecf1', muted:'#b7c0cc', card:'#151821'};

    function apply(theme){

      root.style.setProperty('--bg', theme.bg);

      root.style.setProperty('--bg-alt', theme.bgalt);

      root.style.setProperty('--fg', theme.fg);

      root.style.setProperty('--muted', theme.muted);

      root.style.setProperty('--card', theme.card);

    }

    const saved = localStorage.getItem(LS_KEY);

    if(saved){ apply(saved==='light'?LIGHT:DARK); btn.setAttribute('aria-pressed', saved==='dark'); }

    btn.addEventListener('click', ()=>{

      const isDark = getComputedStyle(root).getPropertyValue('--bg').trim()==='#0b0c10';

      const next = isDark? LIGHT : DARK; 

      apply(next); 

      localStorage.setItem(LS_KEY, isDark? 'light':'dark');

      btn.setAttribute('aria-pressed', !isDark);

    });

  </script>
